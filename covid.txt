import requests
from bs4 import BeautifulSoup 
url  = 'https://covid19dashboard.py.gov.in/BedAvailabilityDetails'
r = requests.get(url) 
soup = requests.get(url) 
tr = soup.find_all('tr',attrs = {'class':'text-right'}) 
ls = []
for i in tr:
    td = i.find_all('td')
    x = []
    for j in td:
       x.append(j.text.strip())
    ls.append(x)
